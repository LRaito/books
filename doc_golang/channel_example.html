channel 使用示例

1.基础用法

nil channel

  • 有缓冲和无缓冲的 channel 的零值都是 nil
  • 往 nil channel 写入数据会永久阻塞
  • 从 nil channel 接收数据会永久阻塞
  • 关闭 nil channel 会 panic 
var ch chan int = nil
ch <- 1 // 错误！这里永久阻塞
var ch chan int = nil
a := <-ch     // 错误！这里永久阻塞
a, ok := <-ch // 错误！这里永久阻塞
var ch chan int = nil
close(ch) // 错误！直接 panic: close of nil channel

close channel

  • 往关闭的 channel 发送数据会 panic
  • 从关闭的 channel 能一直接受到零值 
ch := make(chan int, 1)
close(ch)
ch <- 1 // 错误！直接 panic: send on closed channel
ch := make(chan int, 1)
close(ch)
a := <- ch    // 0
b, ok := <-ch // 0, false

2.阻塞操作：select + default

在某些场景下，你不想让 select 阻塞，而是希望立即返回结果或执行默认逻辑。 应用场景：心跳检测、状态轮询、避免 goroutine 死锁

ch := make(chan int, 1)
ch <- 1

// 非阻塞接收
select {
case data := <-ch:
    fmt.Println("Received:", data)
default:
    fmt.Println("No data available")
}

// 非阻塞发送
select {
case ch <- 2:
    fmt.Println("Sent 2")
default:
    fmt.Println("Channel full, cannot send")
}

3.超时控制：time.After

避免 channel 操作无限期阻塞。这是处理网络请求、I/O 操作的标配。

ch := make(chan string)

go func() {
    time.Sleep(3 * time.Second)
    ch <- "result"
}()

select {
case res := <-ch:
    fmt.Println("Received:", res)
case <-time.After(2 * time.Second):
    fmt.Println("Timeout! No response in 2s")
}

4.关闭 channel 与多路复用

正确关闭 channel

  • 只有发送方应该关闭 channel。
  • 从已关闭的 channel 读取会立即返回零值。
  • 使用 ok 判断 channel 是否已关闭。 
ch := make(chan int, 3)
ch <- 1
ch <- 2
close(ch)

for {
    if data, ok := <-ch; ok {
        fmt.Println(data)
    } else {
        fmt.Println("Channel closed")
        break
    }
}

多路复用（Multiplexing）

使用 select 监听多个 channel，实现事件驱动。

c1, c2 := make(chan string), make(chan string)

go func() { c1 <- "from c1" }()
go func() { c2 <- "from c2" }()

select {
case msg1 := <-c1:
    fmt.Println(msg1)
case msg2 := <-c2:
    fmt.Println(msg2)
}

5.扇入（Fan-in）与 扇出（Fan-out）模式

扇入（Fan-in）：多个 channel 汇聚到一个

func merge(cs ...<-chan int) <-chan int {
    var wg sync.WaitGroup
    out := make(chan int)

    output := func(c <-chan int) {
        defer wg.Done()
        for n := range c {
            out <- n
        }
    }

    wg.Add(len(cs))
    for _, c := range cs {
        go output(c)
    }

    // 所有输入 channel 关闭后，关闭输出 channel
    go func() {
        wg.Wait()
        close(out)
    }()

    return out
}

扇出（Fan-out）：一个 channel 分发给多个 worker

func worker(id int, jobs <-chan int, results chan<- int) {
    for job := range jobs {
        fmt.Printf("Worker %d processing %d\n", id, job)
        time.Sleep(time.Second)
        results <- job * 2
    }
}

func main() {
    jobs := make(chan int, 10)
    results := make(chan int, 10)

    // 启动 3 个 worker
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }

    // 发送任务
    for j := 1; j <= 5; j++ {
        jobs <- j
    }
    close(jobs)

    // 收集结果
    for i := 0; i < 5; i++ {
        <-results
    }
}

6.上下文（Context）与 channel 协同

func doWork(ctx context.Context) <-chan string {
    ch := make(chan string)

    go func() {
        defer close(ch)
        for {
            select {
            case <-ctx.Done():
                fmt.Println("Work canceled:", ctx.Err())
                return
            case ch <- "working...":
                time.Sleep(500 * time.Millisecond)
            }
        }
    }()

    return ch
}

func main() {
    ctx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
    defer cancel()

    for msg := range doWork(ctx) {
        fmt.Println(msg)
    }
    fmt.Println("Done")
}

7.单向 channel：提升类型安全

Go 支持单向 channel 类型，用于函数参数，增强接口清晰度。

  • <-chan T：只读 channel
  • chan<- T：只写 channel 
func producer(out chan<- int) {
    for i := 0; i < 5; i++ {
        out <- i
    }
    close(out)
}

func consumer(in <-chan int) {
    for data := range in {
        fmt.Println(data)
    }
}

func main() {
    ch := make(chan int)
    go producer(ch)
    consumer(ch)
}

8.带缓冲 channel 的高级使用

信号量模式（控制并发数）

使用带缓冲 channel 实现并发控制（类似信号量）。

semaphore := make(chan struct{}, 3) // 最多 3 个并发

var wg sync.WaitGroup
for i := 0; i < 10; i++ {
    wg.Add(1)
    go func(id int) {
        defer wg.Done()
        semaphore <- struct{}{} // 获取信号量
        defer func() { <-semaphore }() // 释放信号量

        fmt.Printf("Worker %d running\n", id)
        time.Sleep(1 * time.Second)
    }(i)
}
wg.Wait()

Pipeline 模式

将多个处理阶段串联成流水线。

func gen(nums ...int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for _, n := range nums {
            out <- n
        }
    }()
    return out
}

func square(in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for n := range in {
            out <- n * n
        }
    }()
    return out
}

func main() {
    // gen -> square -> square
    for n := range square(square(gen(2, 3, 4))) {
        fmt.Println(n) // 输出: 16, 81, 256
    }
}

9.使用 reflect.Select 动态处理多个 channel

channels := []chan int{
    make(chan int),
    make(chan int),
    make(chan int),
}
cases := make([]reflect.SelectCase, len(channels))
for i, ch := range channels {
    cases[i] = reflect.SelectCase{
        Dir:  reflect.SelectRecv,
        Chan: reflect.ValueOf(ch),
    }
}

go func() {
    i := 0
    for {
        switch i % 3 {
        case 0:
            channels[0] <- 0
        case 1:
            channels[1] <- 1
        case 2:
            channels[2] <- 2
        }
        i++
        time.Sleep(time.Second)
    }
}()

for {
    chosen, value, ok := reflect.Select(cases)
    fmt.Printf("Read from channel %d: %v, ok=%v\n", chosen, value, ok)
}

10.构建一个可取消的定时任务调度器

func scheduleTask(ctx context.Context, delay time.Duration, task func()) {
    ticker := time.NewTicker(delay)
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            fmt.Println("Scheduler canceled")
            return
        case <-ticker.C:
            task()
        }
    }
}

func main() {
    ctx, cancel := context.WithCancel(context.Background())
    go scheduleTask(ctx, 1*time.Second, func() {
        fmt.Println("Task executed at", time.Now().Format("15:04:05"))
    })

    time.Sleep(5 * time.Second)
    cancel() // 取消任务
    time.Sleep(1 * time.Second)
}
